**Step 2: Derive data for new components**
Add useMemo hooks after existing calculations (around line 650, near existing useMemo blocks):

```javascript
// Derive unique locations from properties for route visualization
const uniqueLocations = useMemo(() => {
  if (!clientData?.properties) return [];
  const locationSet = new Set();
  clientData.properties
    .filter(p => p.selected)
    .forEach(p => {
      if (p.location) locationSet.add(p.location);
    });
  return Array.from(locationSet);
}, [clientData?.properties]);

// Calculate total nights across all properties
const totalNights = useMemo(() => {
  if (!clientData?.properties) return 0;
  return clientData.properties
    .filter(p => p.selected)
    .reduce((sum, p) => {
      const nights = calculateNights(p.checkIn, p.checkOut);
      return sum + nights;
    }, 0);
}, [clientData?.properties]);

// Get earliest check-in and latest check-out for date range
const dateRange = useMemo(() => {
  if (!clientData?.properties || clientData.properties.length === 0) {
    return { startDate: null, endDate: null };
  }
  const selectedProperties = clientData.properties.filter(p => p.selected);
  if (selectedProperties.length === 0) {
    return { startDate: null, endDate: null };
  }

  let earliestCheckIn = null;
  let latestCheckOut = null;

  selectedProperties.forEach(p => {
    const checkInDate = p.checkIn ? new Date(parseDateString(p.checkIn) + 'T00:00:00') : null;
    const checkOutDate = p.checkOut ? new Date(parseDateString(p.checkOut) + 'T00:00:00') : null;

    if (checkInDate && !isNaN(checkInDate.getTime())) {
      if (!earliestCheckIn || checkInDate < earliestCheckIn) {
        earliestCheckIn = checkInDate;
      }
    }
    if (checkOutDate && !isNaN(checkOutDate.getTime())) {
      if (!latestCheckOut || checkOutDate > latestCheckOut) {
        latestCheckOut = checkOutDate;
      }
    }
  });

  return {
    startDate: earliestCheckIn ? formatDate(earliestCheckIn.toISOString().split('T')[0]) : null,
    endDate: latestCheckOut ? formatDate(latestCheckOut.toISOString().split('T')[0]) : null
  };
}, [clientData?.properties, parseDateString, formatDate]);

// Count stats for QuickStats component
const statsData = useMemo(() => {
  const locationCount = uniqueLocations.length;
  const flightCount = clientData?.flights?.filter(f => f.selected).length || 0;
  const vehicleCount = clientData?.transportation?.filter(t => t.selected).length || 0;
  const includedActivityCount = clientData?.activities?.filter(a => a.selected && a.included_in_base).length || 0;
  const optionalActivityCount = clientData?.activities?.filter(a => a.selected && !a.included_in_base).length || 0;

  return {
    locationCount,
    flightCount,
    vehicleCount,
    includedActivityCount,
    optionalActivityCount
  };
}, [uniqueLocations, clientData?.flights, clientData?.transportation, clientData?.activities]);
```
